import java.util.*;
import java.lang.reflect.*;
import java.lang.annotation.*;

public class Main {
	public static void main(String[] args) throws Exception {
		Persona p = new Persona("Aristoteles", 1500);
		p.setApellido("Filosofez");
		
		System.out.format("nombre: %s %s\n", p.getNombre(), p.getApellido());
		System.out.format("edad  : %d\n", p.getEdad() );
		
		Class c1                   = Persona.class;
		Constructor[] constructors = c1.getConstructors();
		Method[] methods           = c1.getMethods();
		String c1name              = c1.getName();
		
		// informacion general
		System.out.println("\n\n==== Reflection ====");
		System.out.format("Clase: [%s]\n", c1name);
		
		// mostrar constructores
		System.out.println("\nConstructores");
		for(Constructor c : constructors) {
			System.out.format("Constructor > %s\n", c.toGenericString() );
		}
		
		// mostrar campos
		System.out.println("\nCampos");
		for(Method m : methods) {
			// buscar anotaciones
			Annotation[] notes = m.getAnnotations();
			System.out.format("Metodo (%d) > %s\n", notes.length,  m.toGenericString() );
			for(Annotation n : notes) {
				System.out.format("\t > %s\n", n.annotationType().getName() );
			}
		}
	}
}

@Retention(RetentionPolicy.RUNTIME)
@interface MiTest {
}

class Persona {

	}

	}
